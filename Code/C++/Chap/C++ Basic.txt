#include<iostream> // Này là bắt buộc iostream là thư viện cần có để code
using namespace std; // Dòng nay khai báo để sử dụng cout, cin để nhập và xuất
					// Không có dòng trên thì sử dụng scanf, printf để nhập và xuất
int main() // Đối với Dev C++ thì là int main(){ Các câu lệnh của program và . bắt buộc . có return 0;}
		  // Đối với C Free thì là void main(){ Các câu lệnh của program }
		 // Sau mỗi câu lệnh nhớ chấm phẩy (;) trừ: sau #include<...>, sau int (hoặc void) main
		// và sau tên các hàm con
{
	int n; // Khai báo biến: <kiểu dữ liệu> <tên biến>; VD: int a;
		  // Có các kiểu dữ liệu như: int, char, float, double, long, bool ...
	cout<<"Pham Le Huyen Trang\n" ; // Xuất ra nhiều dòng: cout<<"a"<<"b";
	cout<<"Nhap n: ";			   // Các kí tự: xuống dòng "\n", khoảng trắng "\s", tab "\t" 
	cin>>n;						  // Nhập vào.
	cout<<"Ban vua nhap: "<<n;   // Xuất nhiều cả chuỗi lẫn biến ( n )
	return 0;	// End program
}
